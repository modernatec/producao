package br.com.moderna.pnld.view.aba1 
{
	import br.com.moderna.pnld.controller.CapituloController;
	import br.com.moderna.pnld.controller.InfoObject;
	import com.refunk.events.TimelineEvent;
	import flash.display.MovieClip;
	import flash.events.MouseEvent;
	import caurina.transitions.Tweener;
	/**
	 * ...
	 * @author Roberto Ono
	 */
		
	public class Capitulo4 extends CapituloController
	{												
		public function Capitulo4() 
		{			
		}

		
		override public function labelReached(e:TimelineEvent):void {	
			super.labelReached(e);
			
			switch(e.currentLabel) {
				case "info":
					Tweener.addTween(this.main.controlController.tituloAnima, {alpha:0, time:.5, transition:"linear"});
					Tweener.addTween(this.main.controlController.logo, { alpha:0, time:.5, transition:"linear" } );
					this.main.clearInterface();
					
					/*
					 * Fluxo para pegar o crédito dinamicamente. Usar o tipoRetorno = 2
					 */
					var txt:String = new InfoObject().getCreditos(2);
					creditos.creditos_finais.htmlText = txt;
				break;
			}
		}
		
		
		override public function endReached(e:TimelineEvent):void {
			trace("endAnima>>>>");
			super.endReached(e);
			this.main.verNovamente();
		}
		
	}

}
